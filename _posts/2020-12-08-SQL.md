### Join
```sql
-- EQUI JOIN
SELECT ?
FROM ?, ?
WHERE ? = ?;

-- INNER JOIN
SELECT ?
FROM ?, ?
ON ? = ?;

-- NATURAL JOIN
SELECT ?
FROM ? NATURAL JOIN ?
USING (?);

-- NON EQUI JOIN
SELECT ?
FROM ?, ?
WHERE ? BETWEEN ? AND ?;

-- SELF JOIN
SELECT ?
FROM ? a, ? b
WHERE a = b;

-- LEFT OUTER JOIN
SELECT ?
FROM ?, ?
WHERE ? = ?(+);

-- RIGHT OUTER JOIN
SELECT ?
FROM ?, ?
WHERE ?(+) = ?;

-- FULL OUTER JOIN
SELECT ?
FROM ? FULL OUTER JOIN ?
ON ? = ?;
