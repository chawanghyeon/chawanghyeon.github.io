# Answer

```python
n = int(input())
cnt = 0
stack = []
answer = []
is_possible = True

for i in range(n):
    x = int(input())

    while cnt < x:
      cnt += 1
      stack.append(cnt)
      answer.append("+")

    if stack[-1] == x:
        stack.pop()
        answer.append("-")
    else:
        is_possible = False
        break

if is_possible == False:
    print("NO")
else:
    print("\n".join(answer))
```
